VNC server
-VNC server bude v LXC kontajneri
-musime mat otvoreny port 5901, aby sme aj nieco videli :)


INSTALACIA
-----------------------------------
SERVER
  sudo apt-get install tightvncserver
  
  # doinstalovat graficke prostredie - XFCE je nenarocne na zdroje
  sudo apt-get install xfce4

KLIENT
  sudo apt-get install xtightvncviewer


KONFIGURACIA
-------------------------------------
SERVER
1) otvorime nizsie uvedeny subor

  vim .vnc/xstartup

2) upravime ho nasledovne (pridal som iba na zaciatok suboru "startxfce4", aby sa rozbehlo desktopove prostredie pri pripojeni vnc klienta):


#!/bin/sh

#unset SESSION_MANAGER
#unset DBUS_SESSION_BUS_ADDRESS

startxfce4 &

xrdb $HOME/.Xresources
xsetroot -solid grey
#x-terminal-emulator -geometry 80x24+10+10 -ls -title "$VNCDESKTOP Desktop" &
#x-window-manager &
# Fix to make GNOME work
export XKL_XMODMAP_DISABLE=1
/etc/X11/Xsession




3) nasledne restartujeme tightvncserver

tightvncserver -kill :1
tightvncserver




POUZIVANIE
-------------------------------------
SERVER
  # spustenie virtualnej obrazovky (tusim xvfb - pohladat prikaz v historii)
  Xvfb :0 -screen 0 1024x768x16 -ac &
  #DISPLAY=:0 exec startxfce4 &		# toto tam ale asi netreba - OTESTOVAT

  # spustenie VNC servera
  tightvncserver

  # skontrolujeme port, na ktorom tightvncserver pocuva
  # hladat nieco v zmysle "Xtightvnc"
  # napr.:
  # tcp        0      0 0.0.0.0:5901            0.0.0.0:*               LISTEN      6818/Xtightvnc
  # vidime, ze tightvncserver pocuva na porte 5901
  netstat -plant

  # ked chceme vediet cislo displeja, na ktory Xvfb vykresluje
  ps -ef|grep Xvfb

KLIENT
  # pripojenie klienta ku VNC serveru
  xtightvncviewer 158.193.152.45:5901

SERVER
  # vypnutie VNC servera
  tightvncserver -kill :1
