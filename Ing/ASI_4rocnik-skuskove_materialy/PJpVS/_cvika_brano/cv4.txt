void Generuj (int pole[], int dlzka, int a, int b)
{

void Vymena (int *p_a, int *p_b)
{
    int docasna = *p_a;
    *p_a = *p_b;
    *p_b = docasna;

}

void Utried (int pole[], int dlzka)
{
    for(int i =0;i<dlzka;i++)
{
    for(int j=0;j<dlzka;j++)
    {
        if(pole [j] > pole[j+1])
            Vymena(pole+j,pole+(j+1));
    }
    
}
}
