  (user_name VARCHAR2(20), datum DATE, operacia CHAR(1), table_name VARCHAR2(20), id NUMBER);

CREATE OR REPLACE TRIGGER osoba_del
  BEFORE DELETE ON osoba
  REFENCING old AS stary
  FOR EACH ROW
  BEGIN
    INSERT INTO log_table (user_name, datum, operacia, table_name, id)
    SELECT user, sysdate, 'D', 'osoba', :stary.id
    FROM dual;
  END;
