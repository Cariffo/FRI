Je 20 bezcov, 7 rozhodcov, ktory ich hodnotia znamkami z TRIA rozdelenia (0, 6, 5.5), vysledne hodnotenie je priemer z tychto znamok, ale najvyssia a najnizsia znamka neplati, cize v skutocnosti sa priemer robi iba z 5tich. Mali sme zistit pravdepodobnost, ze aspon traja ludia budu mat vysledne hodnotenie vyssie ako 5.7

bezci = array[20];
znamky = array[7];

for (i=0, i<1000, i++) {                        // 1000 replikacii
    for (j=0, j<20, j++) {                          // pre kazdeho bezca
        for (k=0, k<7, k++) {                      // kazdy rozhodca
            znamky[k] = TRIA (0,6,5.5);     // pridelenie siedmych znamok
        }
        znamky.sort();                              // zotriedi znamky podla velkosti
        sucet = 0;
         for (k=1, k<=5, k++) {                  // spocitaju sa znamky okrem min a max
            sucet += znamky[k];
        }
        bezci [ j ] = sucet / 5;                    // kazdemu bezcovi sa priradi priemer
    }
    uspech = 0;
    for (j=0, j<20, j++) {                                                      // spocita sa pocet bezcov ktori mali priemer nad 5.7
        if (bezci[ j ] > 5.7) { uspech += 1; }
    }
    if (uspech >= 3) { asponTraja += 1; }      // ak boli takyto bezci aspon traja
}
pravdepodobnost = asponTraja/1000;                          // pravdepodobnost ze sa najdu aspon traja bezci ktori mali hodnotenie nad 5.7
